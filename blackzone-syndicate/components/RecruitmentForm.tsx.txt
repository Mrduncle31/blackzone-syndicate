import React, { useState } from "react";
import { Button } from "@/components/ui/button";

export default function RecruitmentForm() {
  const [isSubmitting, setIsSubmitting] = useState(false);

  const handleFormSubmit = async (e: React.FormEvent<HTMLFormElement>) => {
    e.preventDefault();
    setIsSubmitting(true);

    const formData = new FormData(e.currentTarget);
    const payload = {
      character: formData.get("character"),
      timezone: formData.get("timezone"),
      activity: formData.get("activity"),
      notes: formData.get("notes") || "None"
    };

    try {
      const res = await fetch("/api/submitApplication", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(payload)
      });

      if (res.ok) {
        alert("Application Transmitted. Fly safe o7");
        (e.target as HTMLFormElement).reset();
      } else {
        alert("Comms error. Contact a Director on Discord.");
      }
    } catch (err) {
      alert("Comms error. Contact a Director on Discord.");
    } finally {
      setIsSubmitting(false);
    }
  };

  return (
    <div className="max-w-2xl mx-auto">
      <div className="text-center mb-16">
        <h2 className="text-5xl font-black italic uppercase tracking-tighter">Join the Syndicate</h2>
        <p className="text-zinc-500 mt-2 font-medium">Recruitment Status: <span className="text-green-500 font-bold uppercase">Open</span></p>
      </div>

      <form className="space-y-6" onSubmit={handleFormSubmit}>
        <div className="grid md:grid-cols-2 gap-6">
          <div className="space-y-2">
            <label className="text-[10px] uppercase font-black text-zinc-500 tracking-widest ml-2">Pilot Name</label>
            <input name="character" className="w-full bg-black border border-zinc-800 p-4 rounded-none focus:border-red-600 outline-none transition-colors" placeholder="EVE CHARACTER NAME" required />
          </div>
          <div className="space-y-2">
            <label className="text-[10px] uppercase font-black text-zinc-500 tracking-widest ml-2">Primary TZ</label>
            <input name="timezone" className="w-full bg-black border border-zinc-800 p-4 rounded-none focus:border-red-600 outline-none transition-colors" placeholder="e.g. 23:00 - 04:00 EVE" required />
          </div>
        </div>

        <div className="space-y-2">
          <label className="text-[10px] uppercase font-black text-zinc-500 tracking-widest ml-2">Combat Role</label>
          <input name="activity" className="w-full bg-black border border-zinc-800 p-4 rounded-none focus:border-red-600 outline-none transition-colors" placeholder="DPS / Logi / Scout / Indyl" required />
        </div>

        <div className="space-y-2">
          <label className="text-[10px] uppercase font-black text-zinc-500 tracking-widest ml-2">Notes / History</label>
          <textarea name="notes" rows={4} className="w-full bg-black border border-zinc-800 p-4 rounded-none focus:border-red-600 outline-none resize-none transition-colors" placeholder="Why are you leaving your current corp?" />
        </div>

        <Button type="submit" disabled={isSubmitting} className="w-full rounded-none bg-white text-black hover:bg-red-600 hover:text-white py-8 text-xl font-black italic uppercase transition-all">
          {isSubmitting ? "Transmitting..." : "Transmit Application"}
        </Button>
      </form>
    </div>
  );
}
